CREATE OR REPLACE PACKAGE BODY cti_otx_pkg AS

    FUNCTION get_config(p_key IN VARCHAR2) RETURN VARCHAR2 IS
        l_val VARCHAR2(4000);
    BEGIN
        SELECT config_value
        INTO l_val
        FROM cti_config
        WHERE config_key = p_key;

        RETURN l_val;
    END;

    PROCEDURE log_run(p_status IN VARCHAR2, p_details IN VARCHAR2) IS
    BEGIN
        INSERT INTO job_run_log(job_name, status, details)
        VALUES ('OTX_SYNC', p_status, SUBSTR(p_details, 1, 4000));
        COMMIT;
    END;

    PROCEDURE sync_ipv4_iocs IS
        l_api_key    VARCHAR2(4000);
        l_base_url   VARCHAR2(4000);
        l_headers    VARCHAR2(32767);
        l_resp       DBMS_CLOUD_TYPES.resp;
        l_body       CLOB;
        l_rows       NUMBER := 0;
    BEGIN
        l_api_key  := get_config('OTX_API_KEY');
        l_base_url := get_config('OTX_BASE_URL');

        l_headers := '{"X-OTX-API-KEY":"' || l_api_key || '"}';

        l_resp := DBMS_CLOUD.SEND_REQUEST(
            credential_name => NULL,
            uri             => l_base_url || '/pulses/subscribed?limit=20&page=1',
            method          => DBMS_CLOUD.METHOD_GET,
            headers         => l_headers
        );

        IF DBMS_CLOUD.GET_RESPONSE_STATUS_CODE(l_resp) <> 200 THEN
            RAISE_APPLICATION_ERROR(
                -20001,
                'OTX HTTP error: ' || DBMS_CLOUD.GET_RESPONSE_STATUS_CODE(l_resp)
            );
        END IF;

        l_body := DBMS_CLOUD.GET_RESPONSE_TEXT(l_resp);

        INSERT INTO cti_otx_iocs (
            indicator_type,
            indicator_value,
            pulse_id,
            pulse_name,
            description,
            threat_score,
            first_seen,
            last_seen,
            is_active,
            source_name,
            raw_json,
            created_at,
            updated_at
        )
        SELECT
            jt.indicator_type,
            jt.indicator_value,
            jt.pulse_id,
            jt.pulse_name,
            jt.pulse_desc,
            50,
            SYSTIMESTAMP,
            SYSTIMESTAMP,
            'Y',
            'OTX',
            l_body,
            SYSTIMESTAMP,
            SYSTIMESTAMP
        FROM JSON_TABLE(
            l_body,
            '$.results[*]'
            COLUMNS (
                pulse_id   VARCHAR2(100)  PATH '$.id',
                pulse_name VARCHAR2(500)  PATH '$.name',
                pulse_desc VARCHAR2(2000) PATH '$.description',
                NESTED PATH '$.indicators[*]'
                COLUMNS (
                    indicator_type  VARCHAR2(50)  PATH '$.type',
                    indicator_value VARCHAR2(200) PATH '$.indicator'
                )
            )
        ) jt
        WHERE jt.indicator_type = 'IPv4'
          AND jt.indicator_value IS NOT NULL
          AND NOT EXISTS (
              SELECT 1
              FROM cti_otx_iocs t
              WHERE t.indicator_type  = jt.indicator_type
                AND t.indicator_value = jt.indicator_value
                AND t.pulse_id        = jt.pulse_id
          );

        l_rows := SQL%ROWCOUNT;

        COMMIT;

        log_run('SUCCESS', 'Inserted IPv4 IOC rows = ' || l_rows);

    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK;
            log_run('FAILED', SQLERRM);
            RAISE;
    END sync_ipv4_iocs;

END cti_otx_pkg;

BEGIN

  EXCEPTION
    WHEN OTHERS THEN

        RAISE;
      END IF;
  END;

        principal_name,
        compartment_id,
        compartment_name,
        resource_name,
        ioc_value,

        pulse_name,
        severity,
        raw_event
    )

        JSON_VALUE(raw_line, '$.data.identity.ipAddress' RETURNING VARCHAR2(100) NULL ON ERROR),

          TO_TIMESTAMP_TZ(
            JSON_VALUE(raw_line, '$.datetime' RETURNING VARCHAR2(100) NULL ON ERROR),

        JSON_VALUE(raw_line, '$.data.resourceName' RETURNING VARCHAR2(4000) NULL ON ERROR),

          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BOTNET%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%C2%' THEN 'HIGH'

          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BRUTE%' THEN 'MEDIUM'
          ELSE 'LOW'
        END,
        raw_line
    FROM AUDIT_LOG_ONE_EXT e
    JOIN CTI_OTX_IOCS c
      ON c.indicator_type = 'IPv4'
     AND c.indicator = JSON_VALUE(e.raw_line, '$.data.identity.ipAddress' RETURNING VARCHAR2(100) NULL ON ERROR)
    WHERE JSON_VALUE(e.raw_line, '$.data.identity.ipAddress' RETURNING VARCHAR2(100) NULL ON ERROR) IS NOT NULL
  ]';

    EXECUTE IMMEDIATE 'DROP TABLE AUDIT_LOG_ONE_EXT PURGE';

CREATE OR REPLACE PROCEDURE PROCESS_FLOW_FILE (
    p_object_name IN VARCHAR2
)
AS
  v_base_uri   VARCHAR2(4000) := 'https://objectstorage.us-ashburn-1.oraclecloud.com/n/ociateam/b/Security_Lake_Flow_Logs/o/ocid1.serviceconnector.oc1.iad.amaaaaaac3adhhqa2hxjhxuvfhtjelxsxc6mpklssrt462z3gdffivmcfdsa/';
  v_file_uri   VARCHAR2(4000);
  v_error_msg  VARCHAR2(4000);
BEGIN
  v_file_uri := v_base_uri || REPLACE(p_object_name, ' ', '%20');

  BEGIN
    EXECUTE IMMEDIATE 'DROP TABLE FLOW_LOG_ONE_EXT PURGE';
  EXCEPTION
    WHEN OTHERS THEN
      IF SQLCODE != -942 THEN
        RAISE;
      END IF;
  END;

  DBMS_CLOUD.CREATE_EXTERNAL_TABLE(
      table_name      => 'FLOW_LOG_ONE_EXT',
      credential_name => 'OCI$RESOURCE_PRINCIPAL',
      file_uri_list   => v_file_uri,
      column_list     => 'raw_line VARCHAR2(32767)',
      field_list      => 'raw_line CHAR(32767)',
      format          => json_object(
          'delimiter'       VALUE '0x''01''',
          'recorddelimiter' VALUE '''\n''',
          'compression'     VALUE 'gzip'
      )
  );

  EXECUTE IMMEDIATE q'[
    INSERT INTO FLOW_ALERTS (
        ALERT_TIME,
        EVENT_TIME,
        SRC_IP,
        DST_IP,
        SRC_PORT,
        DST_PORT,
        PROTOCOL,
        ACTION,
        VNIC_OCID,
        SUBNET_ID,
        COMPARTMENT_ID,
        LOG_STATUS,
        OBJECT_NAME,
        INDICATOR_VALUE,
        INDICATOR_TYPE,
        PULSE_ID,
        PULSE_NAME,
        SEVERITY,
        MATCH_REASON,
        RAW_EVENT,
        CREATED_AT
    )
    SELECT
        SYSTIMESTAMP,
        CAST(
          TO_TIMESTAMP_TZ(
            JSON_VALUE(e.raw_line, '$.datetime' RETURNING VARCHAR2(100) NULL ON ERROR),
            'YYYY-MM-DD"T"HH24:MI:SSTZH:TZM'
          ) AS TIMESTAMP
        ) AS event_time,
        JSON_VALUE(e.raw_line, '$.data.sourceAddress' RETURNING VARCHAR2(100) NULL ON ERROR) AS src_ip,
        JSON_VALUE(e.raw_line, '$.data.destinationAddress' RETURNING VARCHAR2(100) NULL ON ERROR) AS dst_ip,
        JSON_VALUE(e.raw_line, '$.data.sourcePort' RETURNING NUMBER NULL ON ERROR) AS src_port,
        JSON_VALUE(e.raw_line, '$.data.destinationPort' RETURNING NUMBER NULL ON ERROR) AS dst_port,
        JSON_VALUE(e.raw_line, '$.data.protocol' RETURNING VARCHAR2(50) NULL ON ERROR) AS protocol,
        JSON_VALUE(e.raw_line, '$.data.action' RETURNING VARCHAR2(50) NULL ON ERROR) AS action,
        JSON_VALUE(e.raw_line, '$.data.vnicId' RETURNING VARCHAR2(200) NULL ON ERROR) AS vnic_ocid,
        JSON_VALUE(e.raw_line, '$.data.subnetId' RETURNING VARCHAR2(200) NULL ON ERROR) AS subnet_id,
        JSON_VALUE(e.raw_line, '$.data.compartmentId' RETURNING VARCHAR2(200) NULL ON ERROR) AS compartment_id,
        JSON_VALUE(e.raw_line, '$.data.status' RETURNING VARCHAR2(100) NULL ON ERROR) AS log_status,
        :1,
        c.indicator_value,
        c.indicator_type,
        c.pulse_id,
        c.pulse_name,
        CASE
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%MALWARE%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BOTNET%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%C2%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%SCANNER%' THEN 'MEDIUM'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BRUTE%' THEN 'MEDIUM'
          ELSE 'LOW'
        END,
        'Matched source IP from flow log to OTX IPv4 IOC',
        e.raw_line,
        SYSTIMESTAMP
    FROM FLOW_LOG_ONE_EXT e
    JOIN CTI_OTX_IOCS c
      ON c.indicator_type = 'IPv4'
     AND c.indicator_value = JSON_VALUE(e.raw_line, '$.data.sourceAddress' RETURNING VARCHAR2(100) NULL ON ERROR)
    WHERE JSON_VALUE(e.raw_line, '$.data.sourceAddress' RETURNING VARCHAR2(100) NULL ON ERROR) IS NOT NULL

    UNION ALL

    SELECT
        SYSTIMESTAMP,
        CAST(
          TO_TIMESTAMP_TZ(
            JSON_VALUE(e.raw_line, '$.datetime' RETURNING VARCHAR2(100) NULL ON ERROR),
            'YYYY-MM-DD"T"HH24:MI:SSTZH:TZM'
          ) AS TIMESTAMP
        ) AS event_time,
        JSON_VALUE(e.raw_line, '$.data.sourceAddress' RETURNING VARCHAR2(100) NULL ON ERROR) AS src_ip,
        JSON_VALUE(e.raw_line, '$.data.destinationAddress' RETURNING VARCHAR2(100) NULL ON ERROR) AS dst_ip,
        JSON_VALUE(e.raw_line, '$.data.sourcePort' RETURNING NUMBER NULL ON ERROR) AS src_port,
        JSON_VALUE(e.raw_line, '$.data.destinationPort' RETURNING NUMBER NULL ON ERROR) AS dst_port,
        JSON_VALUE(e.raw_line, '$.data.protocol' RETURNING VARCHAR2(50) NULL ON ERROR) AS protocol,
        JSON_VALUE(e.raw_line, '$.data.action' RETURNING VARCHAR2(50) NULL ON ERROR) AS action,
        JSON_VALUE(e.raw_line, '$.data.vnicId' RETURNING VARCHAR2(200) NULL ON ERROR) AS vnic_ocid,
        JSON_VALUE(e.raw_line, '$.data.subnetId' RETURNING VARCHAR2(200) NULL ON ERROR) AS subnet_id,
        JSON_VALUE(e.raw_line, '$.data.compartmentId' RETURNING VARCHAR2(200) NULL ON ERROR) AS compartment_id,
        JSON_VALUE(e.raw_line, '$.data.status' RETURNING VARCHAR2(100) NULL ON ERROR) AS log_status,
        :1,
        c.indicator_value,
        c.indicator_type,
        c.pulse_id,
        c.pulse_name,
        CASE
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%MALWARE%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BOTNET%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%C2%' THEN 'HIGH'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%SCANNER%' THEN 'MEDIUM'
          WHEN UPPER(NVL(c.pulse_name,' ')) LIKE '%BRUTE%' THEN 'MEDIUM'
          ELSE 'LOW'
        END,
        'Matched destination IP from flow log to OTX IPv4 IOC',
        e.raw_line,
        SYSTIMESTAMP
    FROM FLOW_LOG_ONE_EXT e
    JOIN CTI_OTX_IOCS c
      ON c.indicator_type = 'IPv4'
     AND c.indicator_value = JSON_VALUE(e.raw_line, '$.data.destinationAddress' RETURNING VARCHAR2(100) NULL ON ERROR)
    WHERE JSON_VALUE(e.raw_line, '$.data.destinationAddress' RETURNING VARCHAR2(100) NULL ON ERROR) IS NOT NULL
  ]' USING p_object_name, p_object_name;

  UPDATE FLOW_NEW_FILES
  SET PROCESSED_FLAG = 'Y',
      PROCESSED_AT   = SYSTIMESTAMP,
      STATUS         = 'PROCESSED',
      ERROR_MESSAGE  = NULL
  WHERE OBJECT_NAME = p_object_name;

  COMMIT;

  BEGIN
    EXECUTE IMMEDIATE 'DROP TABLE FLOW_LOG_ONE_EXT PURGE';
  EXCEPTION
    WHEN OTHERS THEN
      IF SQLCODE != -942 THEN
        RAISE;
      END IF;
  END;

EXCEPTION
  WHEN OTHERS THEN
    v_error_msg := SUBSTR(SQLERRM, 1, 4000);

    BEGIN

    PROCEDURE write_log(

    FUNCTION get_http_clob(
        p_url         IN VARCHAR2,
        p_method      IN VARCHAR2,
        p_contenttype IN VARCHAR2 DEFAULT NULL,
        p_payload     IN CLOB DEFAULT NULL,
        p_auth_key    IN VARCHAR2 DEFAULT NULL
    ) RETURN CLOB IS
        l_req      UTL_HTTP.req;
        l_resp     UTL_HTTP.resp;
        l_buffer   VARCHAR2(32767);
        l_body     CLOB;
    BEGIN
        UTL_HTTP.SET_WALLET('');

        l_req := UTL_HTTP.BEGIN_REQUEST(p_url, p_method, 'HTTP/1.1');

        UTL_HTTP.SET_HEADER(l_req, 'User-Agent', 'Oracle-ADB-CTI-Loader/1.0');
        UTL_HTTP.SET_HEADER(l_req, 'Accept', 'application/json');

        IF p_contenttype IS NOT NULL THEN
            UTL_HTTP.SET_HEADER(l_req, 'Content-Type', p_contenttype);
        END IF;

        IF p_auth_key IS NOT NULL THEN
            UTL_HTTP.SET_HEADER(l_req, 'Auth-Key', p_auth_key);
        END IF;

        IF p_payload IS NOT NULL THEN
            UTL_HTTP.SET_HEADER(l_req, 'Content-Length', TO_CHAR(DBMS_LOB.GETLENGTH(p_payload)));
            UTL_HTTP.WRITE_TEXT(l_req, p_payload);
        END IF;

        l_resp := UTL_HTTP.GET_RESPONSE(l_req);

        IF l_resp.status_code <> 200 THEN
            DBMS_LOB.CREATETEMPORARY(l_body, TRUE);

            BEGIN
                LOOP
                    UTL_HTTP.READ_TEXT(l_resp, l_buffer, 32767);
                    DBMS_LOB.WRITEAPPEND(l_body, LENGTH(l_buffer), l_buffer);
                END LOOP;
            EXCEPTION
                WHEN UTL_HTTP.END_OF_BODY THEN
                    NULL;
            END;

            UTL_HTTP.END_RESPONSE(l_resp);

            RAISE_APPLICATION_ERROR(
                -20002,
                'HTTP call failed. URL=' || p_url ||
                ' status=' || l_resp.status_code ||
                ' body=' || DBMS_LOB.SUBSTR(l_body, 1000, 1)
            );
        END IF;

        DBMS_LOB.CREATETEMPORARY(l_body, TRUE);

        BEGIN
            LOOP
                UTL_HTTP.READ_TEXT(l_resp, l_buffer, 32767);
                DBMS_LOB.WRITEAPPEND(l_body, LENGTH(l_buffer), l_buffer);
            END LOOP;
        EXCEPTION
            WHEN UTL_HTTP.END_OF_BODY THEN
                NULL;
        END;

        UTL_HTTP.END_RESPONSE(l_resp);
        RETURN l_body;
    EXCEPTION
        WHEN OTHERS THEN
            BEGIN
                UTL_HTTP.END_RESPONSE(l_resp);
            EXCEPTION
                WHEN OTHERS THEN
                    NULL;
            END;
            RAISE;
    END get_http_clob;

    PROCEDURE sync_threatfox IS
        l_url           VARCHAR2(4000);
        l_days          NUMBER;
        l_auth_key      VARCHAR2(4000);
        l_payload       CLOB;
        l_body          CLOB;
        l_query_status  VARCHAR2(50);
        l_rows_merged   NUMBER := 0;
    BEGIN
        l_url      := get_config('ABUSECH_THREATFOX_URL');
        l_auth_key := get_config('THREATFOX_AUTH_KEY');
        l_days     := TO_NUMBER(get_config('THREATFOX_DAYS'));

        IF l_days < 1 OR l_days > 7 THEN
            RAISE_APPLICATION_ERROR(-20004, 'THREATFOX_DAYS must be between 1 and 7');
        END IF;

        l_payload := '{"query":"get_iocs","days":' || l_days || '}';

        l_body := get_http_clob(
            p_url         => l_url,
            p_method      => 'POST',
            p_contenttype => 'application/json',
            p_payload     => l_payload,
            p_auth_key    => l_auth_key
        );

        SELECT jt.query_status
        INTO   l_query_status
        FROM   JSON_TABLE(
                   l_body,
                   '$'
                   COLUMNS (
                       query_status VARCHAR2(50) PATH '$.query_status'
                   )
               ) jt;

        IF l_query_status = 'no_results' THEN
            write_log('CTI_SYNC', 'ABUSECH_THREATFOX', 'SUCCESS', 0, 'No results returned');
            RETURN;
        ELSIF l_query_status <> 'ok' THEN
            RAISE_APPLICATION_ERROR(-20003, 'ThreatFox query_status=' || l_query_status);
        END IF;

        MERGE INTO cti_iocs t
        USING (
            SELECT
                'IPv4' AS indicator_type,
                CASE
                    WHEN INSTR(jt.ioc_value, ':') > 0
                    THEN SUBSTR(jt.ioc_value, 1, INSTR(jt.ioc_value, ':') - 1)
                    ELSE jt.ioc_value
                END AS indicator_value,
                'ABUSECH_THREATFOX' AS source_name,
                'THREATFOX-' || jt.ioc_id AS source_key,
                jt.ioc_id AS pulse_id,
                jt.malware_printable AS pulse_name,
                jt.threat_type_desc AS description,
                jt.confidence_level AS threat_score,
                CASE
                    WHEN jt.first_seen IS NOT NULL
                    THEN TO_TIMESTAMP(REPLACE(jt.first_seen, ' UTC', ''), 'YYYY-MM-DD HH24:MI:SS')
                END AS first_seen,
                CASE
                    WHEN jt.last_seen IS NOT NULL
                    THEN TO_TIMESTAMP(REPLACE(jt.last_seen, ' UTC', ''), 'YYYY-MM-DD HH24:MI:SS')
                    WHEN jt.first_seen IS NOT NULL
                    THEN TO_TIMESTAMP(REPLACE(jt.first_seen, ' UTC', ''), 'YYYY-MM-DD HH24:MI:SS')
                    ELSE SYSTIMESTAMP
                END AS last_seen,
                'Y' AS is_active,
                JSON_OBJECT(
                    'id' VALUE jt.ioc_id,
                    'ioc' VALUE jt.ioc_value,
                    'ioc_type' VALUE jt.ioc_type,
                    'malware_printable' VALUE jt.malware_printable,
                    'threat_type_desc' VALUE jt.threat_type_desc,
                    'confidence_level' VALUE jt.confidence_level,
                    'first_seen' VALUE jt.first_seen,
                    'last_seen' VALUE jt.last_seen
                    RETURNING CLOB
                ) AS raw_json
            FROM JSON_TABLE(
                     l_body,
                     '$.data[*]'
                     COLUMNS (
                         ioc_id            VARCHAR2(100)  PATH '$.id',
                         ioc_value         VARCHAR2(200)  PATH '$.ioc',
                         ioc_type          VARCHAR2(50)   PATH '$.ioc_type',
                         malware_printable VARCHAR2(500)  PATH '$.malware_printable',
                         threat_type_desc  VARCHAR2(2000) PATH '$.threat_type_desc',
                         confidence_level  NUMBER         PATH '$.confidence_level',
                         first_seen        VARCHAR2(50)   PATH '$.first_seen',
                         last_seen         VARCHAR2(50)   PATH '$.last_seen'
                     )
                 ) jt
            WHERE jt.ioc_type = 'ip:port'
              AND jt.ioc_value IS NOT NULL
              AND REGEXP_LIKE(
                    CASE
                        WHEN INSTR(jt.ioc_value, ':') > 0
                        THEN SUBSTR(jt.ioc_value, 1, INSTR(jt.ioc_value, ':') - 1)
                        ELSE jt.ioc_value
                    END,
                    '^[0-9]{1,3}(\.[0-9]{1,3}){3}$'
                  )
        ) s
        ON (t.source_name = s.source_name AND t.source_key = s.source_key)
        WHEN MATCHED THEN
            UPDATE SET
                t.indicator_type  = s.indicator_type,
                t.indicator_value = s.indicator_value,
                t.pulse_id        = s.pulse_id,
                t.pulse_name      = s.pulse_name,
                t.description     = s.description,
                t.threat_score    = s.threat_score,
                t.first_seen      = COALESCE(t.first_seen, s.first_seen),
                t.last_seen       = s.last_seen,
                t.is_active       = s.is_active,
                t.raw_json        = s.raw_json,
                t.updated_at      = SYSTIMESTAMP
        WHEN NOT MATCHED THEN
            INSERT (
                indicator_type, indicator_value, source_name, source_key,
                pulse_id, pulse_name, description, threat_score,
                first_seen, last_seen, is_active, raw_json,
                created_at, updated_at
            )
            VALUES (
                s.indicator_type, s.indicator_value, s.source_name, s.source_key,
                s.pulse_id, s.pulse_name, s.description, s.threat_score,
                s.first_seen, s.last_seen, s.is_active, s.raw_json,
                SYSTIMESTAMP, SYSTIMESTAMP
            );

        l_rows_merged := SQL%ROWCOUNT;
        COMMIT;

        write_log('CTI_SYNC', 'ABUSECH_THREATFOX', 'SUCCESS', l_rows_merged,
                  'ThreatFox sync completed');
    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK;
            write_log('CTI_SYNC', 'ABUSECH_THREATFOX', 'FAILED', 0, SQLERRM);
            RAISE;
    END sync_threatfox;

    PROCEDURE sync_feodo IS
        l_url         VARCHAR2(4000);
        l_body        CLOB;
        l_rows_merged NUMBER := 0;
    BEGIN
        l_url := get_config('ABUSECH_FEODO_URL');

        l_body := get_http_clob(
            p_url    => l_url,
            p_method => 'GET'
        );

        MERGE INTO cti_iocs t
        USING (
            SELECT
                'IPv4' AS indicator_type,
                jt.ip_address AS indicator_value,
                'ABUSECH_FEODO' AS source_name,
                'FEODO-' || jt.ip_address || '-' || jt.port AS source_key,
                'FEODO-' || jt.ip_address || '-' || jt.port AS pulse_id,
                NVL(jt.malware, 'Feodo') AS pulse_name,
                'Botnet C2 | Port: ' || jt.port ||
                ' | AS: ' || NVL(jt.as_name, 'unknown') ||
                ' | Country: ' || NVL(jt.country, 'unknown') AS description,
                85 AS threat_score,
                CASE
                    WHEN jt.first_seen IS NOT NULL
                    THEN TO_TIMESTAMP(jt.first_seen, 'YYYY-MM-DD HH24:MI:SS')
                END AS first_seen,
                CASE
                    WHEN jt.last_online IS NOT NULL
                    THEN TO_TIMESTAMP(jt.last_online, 'YYYY-MM-DD HH24:MI:SS')
                    WHEN jt.first_seen IS NOT NULL
                    THEN TO_TIMESTAMP(jt.first_seen, 'YYYY-MM-DD HH24:MI:SS')
                    ELSE SYSTIMESTAMP
                END AS last_seen,
                CASE
                    WHEN LOWER(jt.status) = 'online' THEN 'Y'
                    ELSE 'N'
                END AS is_active,
                JSON_OBJECT(
                    'ip_address' VALUE jt.ip_address,
                    'port' VALUE jt.port,
                    'status' VALUE jt.status,
                    'as_name' VALUE jt.as_name,
                    'country' VALUE jt.country,
                    'first_seen' VALUE jt.first_seen,
                    'last_online' VALUE jt.last_online,
                    'malware' VALUE jt.malware
                    RETURNING CLOB
                ) AS raw_json
            FROM JSON_TABLE(
                     l_body,
                     '$[*]'
                     COLUMNS (
                         ip_address  VARCHAR2(50)  PATH '$.ip_address',
                         port        VARCHAR2(10)  PATH '$.port',
                         status      VARCHAR2(20)  PATH '$.status',
                         as_name     VARCHAR2(500) PATH '$.as_name',
                         country     VARCHAR2(50)  PATH '$.country',
                         first_seen  VARCHAR2(50)  PATH '$.first_seen',
                         last_online VARCHAR2(50)  PATH '$.last_online',
                         malware     VARCHAR2(200) PATH '$.malware'
                     )
                 ) jt
            WHERE jt.ip_address IS NOT NULL
              AND REGEXP_LIKE(jt.ip_address, '^[0-9]{1,3}(\.[0-9]{1,3}){3}$')
        ) s
        ON (t.source_name = s.source_name AND t.source_key = s.source_key)
        WHEN MATCHED THEN
            UPDATE SET
                t.indicator_type  = s.indicator_type,
                t.indicator_value = s.indicator_value,
                t.pulse_id        = s.pulse_id,
                t.pulse_name      = s.pulse_name,
                t.description     = s.description,
                t.threat_score    = s.threat_score,
                t.first_seen      = COALESCE(t.first_seen, s.first_seen),
                t.last_seen       = s.last_seen,
                t.is_active       = s.is_active,
                t.raw_json        = s.raw_json,
                t.updated_at      = SYSTIMESTAMP
        WHEN NOT MATCHED THEN
            INSERT (
                indicator_type, indicator_value, source_name, source_key,
                pulse_id, pulse_name, description, threat_score,
                first_seen, last_seen, is_active, raw_json,
                created_at, updated_at
            )
            VALUES (
                s.indicator_type, s.indicator_value, s.source_name, s.source_key,
                s.pulse_id, s.pulse_name, s.description, s.threat_score,
                s.first_seen, s.last_seen, s.is_active, s.raw_json,
                SYSTIMESTAMP, SYSTIMESTAMP
            );

        l_rows_merged := SQL%ROWCOUNT;
        COMMIT;

        write_log('CTI_SYNC', 'ABUSECH_FEODO', 'SUCCESS', l_rows_merged,
                  'Feodo sync completed');
    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK;
            write_log('CTI_SYNC', 'ABUSECH_FEODO', 'FAILED', 0, SQLERRM);
            RAISE;
    END sync_feodo;

    PROCEDURE sync_all IS
    BEGIN
        sync_threatfox;
        sync_feodo;
    END sync_all;

END cti_abusech_pkg;
